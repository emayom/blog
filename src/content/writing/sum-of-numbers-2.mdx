### 풀이 1: 투 포인터 - 누적 합

```js
const [N, M, ...arr] = require("fs")
                        .readFileSync("/dev/stdin")
                        .toString()
                        .trim()
                        .split(/\s/)
                        .map(v => +v);

let count = 0;

let left = 0;
let right = 0;
let sum = arr[0];

while(left < arr.length && right < arr.length){
    if(sum === M) count++;

    if(sum >= M){
        sum -= arr[left++]
    } else {
        sum += arr[++right];
    }
}

console.log(count);
```
