### 풀이 1:

```js
function solution(keymap, targets) {
    const map = new Map();

    keymap.forEach(keys => {
        [...keys].forEach((key, j) => {
            const count = j + 1;
            const value = map.get(key);

            if(!value || value && value > count){
                map.set(key, count);
            }
        })
    });

    return targets.map(str => {
        let total = 0;

        for(let i = 0; i < str.length; i++){
            if(!map.has(str[i])) return -1;
            total += map.get(str[i]);
        }

        return total;
    });
}
```
